#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    int mx=dots[0][0],my=dots[0][1],lx=dots[0][0],ly=dots[0][1];
    for(int i=1;i<4;i++){
        mx = min(mx,dots[i][0]);
        lx = max(lx,dots[i][0]);
        my = min(my,dots[i][1]);
        ly = max(ly,dots[i][1]);
    }
    answer=abs((lx-mx)*(ly-my));
    return answer;
}
